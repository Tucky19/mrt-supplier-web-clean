if (raw) {
  const parsed = JSON.parse(raw);

  if (Array.isArray(parsed)) {
    setItems(
      parsed.filter(
        (item): item is QuoteItem =>
          item &&
          typeof item === "object" &&
          typeof item.productId === "string" &&
          typeof item.partNo === "string" &&
          typeof item.qty === "number"
      )
    );
  } else {
    localStorage.removeItem(LS_KEY);
  }
}
